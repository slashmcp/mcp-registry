# Example .env file for MCP Registry Backend
# Copy this to .env and fill in your actual API keys

# Database Configuration
DATABASE_URL="postgresql://postgres:postgres@localhost:5432/mcp_registry"

# Server Configuration
PORT=3001
NODE_ENV=development
CORS_ORIGIN=http://localhost:3000

# Google API Keys
# Get your keys from: https://console.cloud.google.com/apis/credentials
# 
# Option 1: Use ONE key for both services (recommended)
# If your API key has both Generative Language API and Cloud Vision API enabled,
# you can use the same key for both:
GOOGLE_GEMINI_API_KEY=your_google_api_key_here

# Option 2: Use separate keys (if you have different keys for each service)
# GOOGLE_GEMINI_API_KEY=your_gemini_api_key_here
# GOOGLE_VISION_API_KEY=your_vision_api_key_here

# Option 3: Use a generic key name (will work for both)
# GOOGLE_API_KEY=your_google_api_key_here

# Alternative: Use service account for Vision API instead of API key
# Uncomment and set path to your service account JSON file
# GOOGLE_APPLICATION_CREDENTIALS=C:/path/to/service-account-key.json
